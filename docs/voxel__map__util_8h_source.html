52  vec_Vec3i rayTrace(const Vec3f &pt1, const Vec3f &pt2) {
53  Vec3f diff = pt2 - pt1;
54  decimal_t k = 0.8;
55  int max_diff = (diff / res_).lpNorm<Eigen::Infinity>() / k;
56  decimal_t s = 1.0 / max_diff;
57  Vec3f step = diff * s;
58 
59  vec_Vec3i pns;
60  Vec3i prev_pn(-1, -1, -1);
61  for (int n = 1; n < max_diff; n++) {
62  Vec3f pt = pt1 + step * n;
63  Vec3i new_pn = floatToInt(pt);
64  if (isOutSide(new_pn))
65  break;
66  if (new_pn != prev_pn)
67  pns.push_back(new_pn);
68  prev_pn = new_pn;
69  }

71  return pns;
